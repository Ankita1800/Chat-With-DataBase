### 1. Race Condition Prevention
```typescript
// âŒ BEFORE: Runs on every datasets change
useEffect(() => {
  const state = loadAppState();
  if (state && datasets.length > 0) {
    const dataset = datasets.find(d => d.id === state.selectedDatasetId);
    setSelectedDataset(dataset); // OVERWRITES new upload!
  }
}, [datasets]); // â† Triggers after every fetch

// âœ… AFTER: Runs once on mount
useEffect(() => {
  if (datasets.length > 0 && !hasRestoredState) {
    const state = loadAppState();
    const dataset = datasets.find(d => d.id === state.selectedDatasetId);
    selectDataset(dataset);
    setHasRestoredState(true); // â† Blocks re-triggers
  }
}, [datasets, hasRestoredState]);
```

### 2. Atomic State Updates
```typescript
// âŒ BEFORE: State updated in multiple places
setSelectedDataset(dataset);
setColumns(dataset.column_names);  // Separate update
setIsUploaded(true);               // Can get out of sync
saveAppState({ ... });             // Async, can be stale

// âœ… AFTER: Single atomic function
const selectDataset = (dataset) => {
  setSelectedDataset(dataset);
  setColumns(dataset?.column_names || []);
  setIsUploaded(dataset !== null);
  if (dataset) {
    saveAppState({
      selectedDatasetId: dataset.id,
      columns: dataset.column_names,
      isUploaded: true,
    });
  }
};
```
